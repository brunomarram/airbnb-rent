def getPrice(price):
    if(price <= 200):

    elif(price <= 500):
        return 1

        return 2
    elif(price <= 900):
        return 3
    elif(price <= 2000):
        return 4
    elif(price <= 4000):
        return 5
    elif(price <= 6000):
        return 6
    elif(price > 6000):
        return 7
